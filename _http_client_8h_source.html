123  template<typename T>
124  T
125  onRequest (web::http::http_response response)
126  {
127  auto headers = response.headers();
128  auto ctype = headers.find(U("Content-Type"));
129  auto jsonType = utility::string_t(U("application/json"));
130  if (ctype != headers.end() && ctype->second.compare(0, jsonType.size(), jsonType) == 0)
131  {
132  auto json = response.extract_json(true).get();
133  auto s = JSonUnserializer{json};
134  if (response.status_code() == 200)
135  {
136  try
137  {
138  return s.unserialize<T>();
139  }
140  catch (const std::exception& e)
141  {
142  GIGA_DEBUG_LOG(U("Error unserializing: ") << json.serialize());
143  throw e;
144  }
145  }
146  else
147  {
148  throwHttpError(response.status_code(), std::move(json));
149  }
150  }
151  BOOST_THROW_EXCEPTION((HttpErrorGeneric{response.status_code(), response.extract_string().get()}));
152  }
